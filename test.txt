#import "basic.txt"

extern 
{
	add_module(string, string) : number
	compile_module(number) : void
	run_module(number) : void
}

var madd : number = add_module("add", "write module_args[0] + module_args[1]")

compile_module(madd)

func add(x : number, y : number) : void
{
	module_args = [x, y]
	run_module(madd)
}

for var i : number = 0, i < 10, i = i + 1
	add(10, 20)
