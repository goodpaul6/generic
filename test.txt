extern iup_label(string) : native
extern iup_vbox(array-native) : native
extern iup_dialog(native) : native
extern iup_set_attribute(native, string, string) : void
extern iup_show_xy(native, number, number) : void
extern iup_main_loop() : void
extern iup_close() : void

struct label
{
	handle : native

	make(name : string) : void
	{
		self.handle = iup_label(name)
	}
}

var label : label = new label
label:make("hello")

var label2 : native = iup_label("This is a label.")
var dlg : native = iup_dialog(iup_vbox([label.handle, label2]))
iup_set_attribute(dlg, "TITLE", "Hello World 2")

iup_show_xy(dlg, 100, 100)

iup_main_loop()

iup_close()
